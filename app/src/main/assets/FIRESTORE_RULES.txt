/**
 * Firestore Security Rules for FitMind AI Coach
 * 
 * Copy these rules to Firebase Console → Firestore Database → Rules tab
 * 
 * These rules ensure:
 * - Only authenticated users can access data
 * - Users can only read/write their own data
 * - Proper data isolation between users
 */

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Helper function to check if user is authenticated
    function isAuthenticated() {
      return request.auth != null;
    }
    
    // Helper function to check if user owns the resource
    function isOwner(userId) {
      return isAuthenticated() && request.auth.uid == userId;
    }
    
    // User profile documents
    match /users/{userId} {
      // Users can read and write their own profile
      allow read, write: if isOwner(userId);
      
      // Workouts subcollection
      match /workouts/{workoutId} {
        // Users can CRUD their own workouts
        allow read: if isOwner(userId);
        allow create: if isOwner(userId) 
                      && request.resource.data.keys().hasAll(['type', 'startTime', 'duration', 'avgHR', 'maxHR', 'calories']);
        allow update, delete: if isOwner(userId);
      }
      
      // Chat sessions subcollection
      match /chats/{chatId} {
        // Users can CRUD their own chats
        allow read: if isOwner(userId);
        allow create: if isOwner(userId) 
                      && request.resource.data.keys().hasAll(['id', 'title', 'messages', 'createdAt']);
        allow update, delete: if isOwner(userId);
      }
    }
    
    // Deny all other access by default
    match /{document=**} {
      allow read, write: if false;
    }
  }
}

/**
 * IMPORTANT NOTES:
 * 
 * 1. Data Structure:
 *    users/{userId}/
 *      - User profile and preferences
 *      workouts/{workoutId}/
 *        - Individual workout data
 *      chats/{chatId}/
 *        - Chat session data with messages
 * 
 * 2. Required Fields:
 *    Workouts must have: type, startTime, duration, avgHR, maxHR, calories
 *    Chats must have: id, title, messages, createdAt
 * 
 * 3. Security:
 *    - Users can only access their own data (userId must match auth.uid)
 *    - All other access is explicitly denied
 *    - No public read/write access
 * 
 * 4. Testing Rules:
 *    Use Firebase Console → Firestore → Rules Playground to test:
 *    - Authenticated user reading own data → ✅ Allow
 *    - Unauthenticated user reading → ❌ Deny
 *    - User reading another user's data → ❌ Deny
 *    - User writing to their own collection → ✅ Allow
 */
